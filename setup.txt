# 1. Crear entorno virtual si no existe
python -m venv env

# 2. Activar entorno virtual
env\Scripts\activate

# 3. Instalar dependencias
pip install --upgrade pip
pip install -r requirements.txt

# 4. (IMPORTANTE) Crear/actualizar la base de datos MySQL
#    - Conectarse a MySQL y ejecutar el script SQL de tablas:
#      -> script_schema_pepsico.sql  (recinto, empleados, vehiculos, ordenestrabajo, pausas, etc.)
#    La BD se administra fuera de Django (managed = False en la mayor√≠a de los modelos).

#Modificar .env con credenciales de BD
# 5. Aplicar migraciones de Django (solo apps internas + chat + documentos)
python manage.py migrate

# 6. Crear superusuario Django (para acceder a /admin)
python manage.py createsuperuser

# 7. Levantar servidor de desarrollo
python manage.py runserver
